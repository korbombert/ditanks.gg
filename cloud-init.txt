'Smasher': { barrels: [], dmg: 1, spd: 1, rel: 1, maxDrones: 0, isSmasher: true }

// --- CLASSES ---
class Room {
constructor(id, mode) {
this.id = id;
this.mode = mode;
this.entities = [];
this.bullets = [];
this.drones = [];
this.clients = [];
this.nextEntityId = 1;
this.spawnEntities();
}

spawnEntities() {
for(let i=0; i<200; i++) {
this.spawnShape();
}
for(let i=0; i<20; i++) this.spawnBot(i);
}

spawnShape() {
let x = Math.random() * WORLD_SIZE, y = Math.random() * WORLD_SIZE;
let isCenter = x > 1200 && x < 1800 && y > 1200 && y < 1800;
let type;

if (isCenter) {
type = ['pentagon', 'pentagon', 'hexagon', 'triangle'][Math.floor(Math.random()*4)];
} else {
type = ['square','square','square','triangle'][Math.floor(Math.random()*4)];
}
this.entities.push(new Entity(this, x, y, type));
}

spawnBot(index, startScore = 0) {
let team = this.mode === "2TDM" ? (index % 2 === 0 ? 1 : 2) : index + 5;
let bx, by;

if(this.mode === "2TDM") {
bx = team === 1 ? Math.random()*200 + 450 : WORLD_SIZE - 450 - Math.random()*200;
by = team === 1 ? Math.random()*200 + 450 : WORLD_SIZE - 450 - Math.random()*200;
} else {
bx = Math.random() * WORLD_SIZE;
by = Math.random() * WORLD_SIZE;
}
let bot = new Entity(this, bx, by, 'ai', "", team);
if (startScore > 0) bot.addXP(startScore);
this.entities.push(bot);
}
}

class Entity {
constructor(room, x, y, type, name = "", team = 0, isPlayer = false, ws = null) {
this.room = room;
this.id = room.nextEntityId++;
this.x = x; this.y = y; this.type = type;
this.name = name; this.team = team;
this.isPlayer = isPlayer; this.ws = ws;
this.tankType = 'Basic';
this.score = 0; this.level = 1; this.xp = 0; this.statPoints = 0;
this.stats = [0,0,0,0,0,0,0,0];
this.hp = 100; this.maxHp = 100;
this.radius = 20; this.angle = 0;
this.vx = 0; this.vy = 0;
this.reloadTimer = 0; this.activeDrones = 0;
this.inputs = { up: false, down: false, left: false, right: false, shooting: false, angle: 0, repel: false };
this.markedForDeletion = false;
this.spawnTime = Date.now();
this.targetId = null;
this.lastDamagedBy = null;

if(type === 'square') { this.hp = 15; this.maxHp = 15; this.radius = 15; this.xpVal = 60; }
if(type === 'triangle') { this.hp = 35; this.maxHp = 35; this.radius = 18; this.xpVal = 150; }
if(type === 'pentagon') { this.hp = 120; this.maxHp = 120; this.radius = 30; this.xpVal = 600; }
if(type === 'hexagon') { this.hp = 350; this.maxHp = 350; this.radius = 45; this.xpVal = 2000; }
if(type === 'ai') { this.name = BOT_NAMES[Math.floor(Math.random()*BOT_NAMES.length)]; }
}

addXP(amt) {
if(['square','triangle','pentagon','hexagon'].includes(this.type)) return;
this.score += amt;

if(this.level >= MAX_LEVEL) return;

this.xp += amt;
let req = this.level * 40;
while(this.xp >= req && this.level < MAX_LEVEL) {
this.level++; this.xp -= req; this.statPoints++;
this.maxHp += 15; this.hp = this.maxHp;
this.radius = 20 + (this.level * 0.3);
req = this.level * 40;
}
if(this.level === MAX_LEVEL) this.xp = 0;
}

update() {
this.vx *= 0.85; this.vy *= 0.85;
this.x += this.vx; this.y += this.vy;
this.x = Math.max(0, Math.min(WORLD_SIZE, this.x));
this.y = Math.max(0, Math.min(WORLD_SIZE, this.y));

if(this.reloadTimer > 0) this.reloadTimer--;
if(this.hp < this.maxHp) this.hp += 0.02 * (1 + this.stats[0]*0.5);

let moveSpeed = 0.5 + (this.stats[7] * 0.04);

if(this.isPlayer) {
if(this.inputs.up) this.vy -= moveSpeed;
if(this.inputs.down) this.vy += moveSpeed;
if(this.inputs.left) this.vx -= moveSpeed;
if(this.inputs.right) this.vx += moveSpeed;
this.angle = this.inputs.angle;
if(this.inputs.shooting) shoot(this);
} else if(this.type === 'ai') {
let evadeVx = 0, evadeVy = 0, evadeCount = 0;
this.room.bullets.forEach(b => {
if (b.owner === this) return;
let isSameTeam = this.room.mode === "2TDM" && b.team === this.team && b.team !== 0;
if (isSameTeam) return;
let dx = b.x - this.x, dy = b.y - this.y;
let d = Math.hypot(dx, dy);
if (d < 250) {
let dot = (b.vx * dx + b.vy * dy);
if (dot < 0) {
evadeVx += (-dy/d) * moveSpeed;
evadeVy += (dx/d) * moveSpeed;
evadeCount++;
}
}
});
if (evadeCount > 0) {
this.vx += evadeVx / evadeCount;
this.vy += evadeVy / evadeCount;
}

let enemyTarget = null; let minEnemyDist = Infinity;
let shapeTarget = null; let minShapeDist = Infinity;
let droneTarget = null; let minDroneDist = Infinity;

// Target Drones
this.room.drones.forEach(d => {
if (d.owner === this || d.markedForDeletion) return;
let isSameTeam = this.room.mode === "2TDM" && d.team === this.team && d.team !== 0;
if (isSameTeam) return;
let dDist = Math.hypot(this.x - d.x, this.y - d.y);
if (dDist < 600 && dDist < minDroneDist) { minDroneDist = dDist; droneTarget = d; }
});

// Target Entities
this.room.entities.forEach(e => {
if(e === this || e.markedForDeletion) return;

let isTank = ['tank', 'ai'].includes(e.type);
if (isTank && (this.score < 1000 || e.score < 1000)) return;

let isSameTeam = this.room.mode === "2TDM" && e.team === this.team && e.team !== 0;
let isShape = ['square','triangle','pentagon','hexagon'].includes(e.type);
if (isSameTeam && !isShape) return;

let d = Math.hypot(this.x - e.x, this.y - e.y);

if (!isShape && !isSameTeam) {
if(d < 600 && d < minEnemyDist) { minEnemyDist = d; enemyTarget = e; }
} else if (isShape) {
if(d < minShapeDist) { minShapeDist = d; shapeTarget = e; }
}
});

// Priority: Drones <- Players <- Shapes
let target = droneTarget || enemyTarget || shapeTarget;
this.targetId = target ? (target.id || null) : null;

if(target) {
let dTarget = Math.hypot(target.x - this.x, target.y - this.y);
let bSpd = (6 + (this.stats[3]*0.4)) * (TANK_SPECS[this.tankType].spd || 1);
let timeToHit = dTarget / (bSpd || 1);
let predX = target.x + (target.vx || 0) * timeToHit;
let predY = target.y + (target.vy || 0) * timeToHit;

this.angle = Math.atan2(predY - this.y, predX - this.x);
shoot(this);

let keepDist = this.tankType === 'Smasher' ? 0 : (this.tankType === 'Sniper' ? 300 : (enemyTarget ? 150 : 50));
let d = enemyTarget ? minEnemyDist : minShapeDist;

if(d > keepDist) {
this.vx += Math.cos(this.angle) * moveSpeed;
this.vy += Math.sin(this.angle) * moveSpeed;
} else if (d < keepDist - 50) {
this.vx -= Math.cos(this.angle) * moveSpeed;
this.vy -= Math.sin(this.angle) * moveSpeed;
}
} else {
this.vx += Math.cos(this.angle) * (moveSpeed * 0.4);
this.vy += Math.sin(this.angle) * (moveSpeed * 0.4);
if(Math.random() < 0.02) this.angle += (Math.random() - 0.5);
}

if(this.level >= 15 && this.tankType === 'Basic') {
const choices = ['Twin', 'Sniper', 'Machine Gun', 'Flank Guard','Overseer', 'Smasher'];
this.tankType = choices[Math.floor(Math.random() * choices.length)];
}
}

if(this.type === 'ai' && this.statPoints > 0){
const bulletStats = [3,4,5,6];
const otherStats = [0,1,2,7];
let pool = [];
bulletStats.forEach(s=>{ if(this.stats[s] < 7) for(let i=0;i<3;i++) pool.push(s); });
otherStats.forEach(s=>{ if(this.stats[s] < 7) pool.push(s); });

if(pool.length){
let choice = pool[Math.floor(Math.random()*pool.length)];
this.stats[choice]++;
this.statPoints--;
}
}
}
}

class Drone {
constructor(room, x, y, owner) {
this.room = room;
this.x = x; this.y = y; this.owner = owner;
this.vx = 0; this.vy = 0; this.radius = 12;
this.hp = 20 + (owner.stats[4] * 5);
this.dmg = 15 + (owner.stats[5] * 5);
this.team = owner.team; this.angle = 0;
this.markedForDeletion = false;
owner.activeDrones++;
}

update() {
let speed = 4 + (this.owner.stats[3] * 0.5);
let tx = this.x, ty = this.y, moving = false;

if (this.owner.isPlayer) {
tx = this.owner.inputs.mouseX; ty = this.owner.inputs.mouseY; moving = true;
if(this.owner.inputs.repel) {
let ang = Math.atan2(this.y - ty, this.x - tx);
tx = this.x + Math.cos(ang) * 500; ty = this.y + Math.sin(ang) * 500;
}
} else {
let target = null; let minDist = Infinity;
this.room.entities.forEach(e => {
if(e === this.owner || e.markedForDeletion) return;
if(this.room.mode === "2TDM" && e.team === this.owner.team && ['ai','tank'].includes(e.type)) return;
let d = Math.hypot(this.x - e.x, this.y - e.y);
if(d < 800 && d < minDist) { minDist = d; target = e; }
});
if(target) { tx = target.x; ty = target.y; moving = true; }
else { tx = this.owner.x; ty = this.owner.y; moving = true; }
}

if (moving) {
let angle = Math.atan2(ty - this.y, tx - this.x);
this.angle = angle;
this.vx += Math.cos(angle) * 0.5; this.vy += Math.sin(angle) * 0.5;
}

let curSpeed = Math.hypot(this.vx, this.vy);
if(curSpeed > speed) {
this.vx = (this.vx / curSpeed) * speed;
this.vy = (this.vy / curSpeed) * speed;
}

this.x += this.vx; this.y += this.vy;

this.room.drones.forEach(d => {
if(d !== this && !d.markedForDeletion) {
let dist = Math.hypot(this.x - d.x, this.y - d.y);
if(dist < this.radius * 2) {
let pushAngle = Math.atan2(this.y - d.y, this.x - d.x);
this.x += Math.cos(pushAngle) * 1; this.y += Math.sin(pushAngle) * 1;
}
}
});
}
}

const rooms = {
"FFA-S1": new Room("FFA-S1", "FFA"),
"FFA-S2": new Room("FFA-S2", "FFA"),
"2TDM-S1": new Room("2TDM-S1", "2TDM"),
"2TDM-S2": new Room("2TDM-S2", "2TDM")
};

function shoot(who) {
if(who.reloadTimer > 0 || who.markedForDeletion) return;
const specs = TANK_SPECS[who.tankType];
const room = who.room;

if(specs.isSmasher) return;

if(specs.isDroneSpawner) {
if(who.activeDrones < specs.maxDrones) {
room.drones.push(new Drone(room, who.x, who.y, who));
who.reloadTimer = Math.max(10, (40 - who.stats[6]*4) * specs.rel);
}
return;
}

let bSpeed = (6 + (who.stats[3]*0.4)) * specs.spd;
let bDmg = (8 + (who.stats[5]*3)) * specs.dmg;
let bPen = 1 + (who.stats[4]*0.5);

specs.barrels.forEach(b => {
let spreadAngle = (Math.random() - 0.5) * b.spread;
let finalAngle = who.angle + b.angle + spreadAngle;
let bx = who.x + Math.cos(who.angle + b.angle) * b.l * who.radius - Math.sin(who.angle) * b.y;
let by = who.y + Math.sin(who.angle + b.angle) * b.l * who.radius + Math.cos(who.angle) * b.y;

let recoilForce = (1.5 + bDmg * 0.05) * specs.dmg;
who.vx -= Math.cos(finalAngle) * recoilForce * 0.25;
who.vy -= Math.sin(finalAngle) * recoilForce * 0.25;

room.bullets.push({
x: bx, y: by, vx: Math.cos(finalAngle)*bSpeed, vy: Math.sin(finalAngle)*bSpeed,
r: 8 + (who.stats[5]*0.5), life: 100 * (1 + who.stats[3]*0.1), dmg: bDmg, pen: bPen,
owner: who, team: who.team
});
});

who.reloadTimer = Math.max(5, (30 - who.stats[6]*3) * specs.rel);
}

// --- MAIN GAME LOOP ---
function updateRoom(room) {
if(room.mode === "2TDM") {
room.entities.forEach(e => {
if(!['tank','ai'].includes(e.type) || e.markedForDeletion) return;

if(e.team !== 1 && e.x < BASE_SIZE && e.y < BASE_SIZE) {
if (e.x > e.y) e.y = BASE_SIZE; else e.x = BASE_SIZE;
}
if(e.team !== 2 && e.x > WORLD_SIZE-BASE_SIZE && e.y > WORLD_SIZE-BASE_SIZE) {
if ((WORLD_SIZE-BASE_SIZE) - e.x > (WORLD_SIZE-BASE_SIZE) - e.y) e.y = WORLD_SIZE-BASE_SIZE;
else e.x = WORLD_SIZE-BASE_SIZE;
}
});
}

for (let i = 0; i < room.bullets.length; i++) {
let b1 = room.bullets[i];
if (b1.life <= 0) continue;
for (let j = i + 1; j < room.bullets.length; j++) {
let b2 = room.bullets[j];
if (b2.life <= 0) continue;

let sameOwner = b1.owner === b2.owner;
let sameTeam = room.mode === "2TDM" && b1.team === b2.team && b1.team !== 0;

if (!sameOwner && !sameTeam) {
if (Math.hypot(b1.x - b2.x, b1.y - b2.y) < b1.r + b2.r) {
let p1 = b1.pen; let p2 = b2.pen;
b1.pen -= p2; b2.pen -= p1;
if (b1.pen <= 0) b1.life = 0;
if (b2.pen <= 0) b2.life = 0;
}
}
}
}

for (let i = 0; i < room.drones.length; i++) {
let d1 = room.drones[i];
if (d1.markedForDeletion) continue;
for (let j = i + 1; j < room.drones.length; j++) {
let d2 = room.drones[j];
if (d2.markedForDeletion) continue;

let sameOwner = d1.owner === d2.owner;
let sameTeam = room.mode === "2TDM" && d1.team === d2.team && d1.team !== 0;

if (!sameOwner && !sameTeam) {
if (Math.hypot(d1.x - d2.x, d1.y - d2.y) < d1.radius + d2.radius) {
let d1Dmg = d1.dmg; let d2Dmg = d2.dmg;
d1.hp -= d2Dmg; d2.hp -= d1Dmg;
if (d1.hp <= 0) d1.markedForDeletion = true;
if (d2.hp <= 0) d2.markedForDeletion = true;
}
}
}
}

for(let i=room.bullets.length-1; i>=0; i--) {
let b = room.bullets[i];

if (room.mode === "2TDM") {
if (b.team !== 1 && b.x < BASE_SIZE && b.y < BASE_SIZE) b.life = 0;
if (b.team !== 2 && b.x > WORLD_SIZE-BASE_SIZE && b.y > WORLD_SIZE-BASE_SIZE) b.life = 0;
}

b.x += b.vx; b.y += b.vy; b.life--;
if(b.life <= 0) { room.bullets.splice(i, 1); continue; }

let hit = false;
room.drones.forEach(d => {
if(hit || d.markedForDeletion || b.owner === d.owner) return;
let isSameTeam = room.mode === "2TDM" && d.team === b.team && b.team !== 0;
if(isSameTeam) return;

if(Math.hypot(b.x - d.x, b.y - d.y) < b.r + d.radius) {
d.hp -= b.dmg;
b.pen -= 1;
if(b.pen <= 0) hit = true;
}
});
room.entities.forEach(en => {
if(hit || en.markedForDeletion || b.owner === en) return;
if(room.mode === "2TDM" && en.team === b.team && !['square','triangle','pentagon','hexagon'].includes(en.type)) return;

if(Math.hypot(b.x - en.x, b.y - en.y) < b.r + en.radius) {
en.hp -= b.dmg; b.pen -= 1;
if(b.owner) en.lastDamagedBy = b.owner.id;
if(b.pen <= 0) hit = true;

if(en.hp <= 0 && b.owner && typeof b.owner.addXP === 'function') {
let xpGain = ['tank','ai'].includes(en.type) ? Math.max(en.xpVal||100, en.score) : (en.xpVal || 100);
b.owner.addXP(Math.min(xpGain,24700));
}
}
});
if(hit) room.bullets.splice(i, 1);
}

for(let i=room.drones.length-1; i>=0; i--) {
let d = room.drones[i];
if (d.markedForDeletion) continue;
d.update();
let hit = false;

room.entities.forEach(en => {
if(hit || en.markedForDeletion || d.owner === en) return;
if(room.mode === "2TDM" && en.team === d.team && !['square','triangle','pentagon','hexagon'].includes(en.type)) return;

if(Math.hypot(d.x - en.x, d.y - en.y) < d.radius + en.radius) {
en.hp -= d.dmg; d.hp -= ['ai','tank'].includes(en.type) ? 20 : 5;
if(d.owner) en.lastDamagedBy = d.owner.id;
if(d.hp <= 0) hit = true;

if(en.hp <= 0 && d.owner && typeof d.owner.addXP === 'function') {
let xpGain = ['tank','ai'].includes(en.type) ? Math.max(en.xpVal||100, en.score) : (en.xpVal || 100);
d.owner.addXP(xpGain);
}
}
});
if(hit || d.owner.hp <= 0 || d.owner.markedForDeletion) {
if(d.owner) d.owner.activeDrones--;
d.markedForDeletion = true;
}
}
room.drones = room.drones.filter(d => !d.markedForDeletion);

room.entities.forEach(en => {
if(en.markedForDeletion) return;
en.update();
if(['square','triangle','pentagon','hexagon'].includes(en.type)) {
en.angle += (en.type==='square'?0.01:(en.type==='triangle'?0.02:0.005));
}

room.entities.forEach(other => {
if(en === other || other.markedForDeletion || en.markedForDeletion) return;
let isSameTeam = room.mode === "2TDM" && en.team === other.team && en.team !== 0;

if(Math.hypot(en.x - other.x, en.y - other.y) < en.radius + other.radius) {
let ang = Math.atan2(other.y - en.y, other.x - en.x);

en.vx -= Math.cos(ang)*2; en.vy -= Math.sin(ang)*2;
other.vx += Math.cos(ang)*2; other.vy += Math.sin(ang)*2;

if (!isSameTeam) {
let dmgE = 1 + (other.stats ? other.stats[2]*2 : 0);
let dmgO = 1 + (en.stats ? en.stats[2]*2 : 0);

en.hp -= dmgE; other.hp -= dmgO;
en.lastDamagedBy = other.id; other.lastDamagedBy = en.id;

if(en.hp <= 0 && typeof other.addXP === 'function') other.addXP(['tank','ai'].includes(en.type) ? Math.max(en.xpVal||100, en.score) : (en.xpVal || 100));
if(other.hp <= 0 && typeof en.addXP === 'function') en.addXP(['tank','ai'].includes(other.type) ? Math.max(other.xpVal||100, other.score) : (other.xpVal || 100));
}
}
});
});

let respawningBots = [];

room.entities = room.entities.filter(e => {
if(e.hp <= 0) {
if(e.isPlayer && e.ws) {
let timeAlive = Math.floor((Date.now() - e.spawnTime) / 1000);
e.ws.send(JSON.stringify({ type: 'death', score: e.score, level: e.level, timeAlive: timeAlive, tank: e.tankType, killerId: e.lastDamagedBy }));

// Track spectating logic
let client = room.clients.find(c => c.player === e);
if (client) {
client.player = null;
client.spectatingId = e.lastDamagedBy;
}
}
if(e.type === 'ai') {
respawningBots.push(Math.min(e.score / 5, 6124));
}

// Move chain spectators to the killer
room.clients.forEach(c => {
if (c.spectatingId === e.id) {
c.spectatingId = e.lastDamagedBy || null;
}
});

return false;
}
return true;
});

respawningBots.forEach(score => room.spawnBot(Math.floor(Math.random()*100), score));

let shapes = room.entities.filter(e => !['ai','tank'].includes(e.type)).length;
if(shapes < 200) room.spawnShape();
}

dctx.strokeRect(0, -b.w/2 + (b.y||0), r * b.l, b.w);
}
dctx.restore();
});

if (data.tank === 'Smasher') {
dctx.fillStyle = "#555"; dctx.strokeStyle = darkenColor("#555", 30);
dctx.beginPath();
for(let i=0; i<6; i++){
let a = (i * Math.PI * 2 / 6);
dctx.lineTo(Math.cos(a)*r*1.4, Math.sin(a)*r*1.4);

function getTeamColor(team) {
if(team === 0) return "#fff";
if(gameMode === "FFA") return team === myTeam ? COLORS.team1 : COLORS.team2;
return team === 1 ? COLORS.team1 : COLORS.team2;
}

function drawPoly(sides, r) {
ctx.beginPath();
for(let i=0; i<sides; i++){
let a = (i * Math.PI * 2 / sides);
ctx.lineTo(Math.cos(a)*r*1.2, Math.sin(a)*r*1.2);
}
ctx.closePath(); ctx.fill(); ctx.stroke();
}

function draw() {
ctx.fillStyle = COLORS.bg; ctx.fillRect(0,0,canvas.width,canvas.height);

let targetCamX = WORLD_SIZE/2 - canvas.width/2;
let targetCamY = WORLD_SIZE/2 - canvas.height/2;

if (myId) {
let me = gameState.entities.find(e => e.id === myId);
if(me) {
if (!renderEntities.has(me.id)) renderEntities.set(me.id, { x: me.x, y: me.y });
let rMe = renderEntities.get(me.id);
targetCamX = rMe.x - canvas.width/2;
targetCamY = rMe.y - canvas.height/2;

if (myStats.tankType === 'Sniper' && (mouse.rightDown || keys.shift)) {
let mouseAngle = Math.atan2(mouse.ry - rMe.y, mouse.rx - rMe.x);
targetCamX += Math.cos(mouseAngle) * 450;
targetCamY += Math.sin(mouseAngle) * 450;
}
}
} else if (spectateId) {
let spec = gameState.entities.find(e => e.id === spectateId);
if(spec) {
if (!renderEntities.has(spec.id)) renderEntities.set(spec.id, { x: spec.x, y: spec.y });
let rSpec = renderEntities.get(spec.id);
targetCamX = rSpec.x - canvas.width/2;
targetCamY = rSpec.y - canvas.height/2;
}
}

camera.x += (targetCamX - camera.x) * 0.1;
camera.y += (targetCamY - camera.y) * 0.1;

mouse.rx = mouse.x + camera.x;
mouse.ry = mouse.y + camera.y;

ctx.strokeStyle = "rgba(0,0,0,0.06)";
ctx.lineWidth = 1;
ctx.beginPath();
for(let x=-camera.x%50; x<canvas.width; x+=50) { ctx.moveTo(x,0); ctx.lineTo(x,canvas.height); }
for(let y=-camera.y%50; y<canvas.height; y+=50) { ctx.moveTo(0,y); ctx.lineTo(canvas.width,y); }
ctx.stroke();

if(gameMode === "2TDM") {
ctx.fillStyle = "rgba(0, 178, 225, 0.15)"; ctx.fillRect(-camera.x, -camera.y, BASE_SIZE, BASE_SIZE);
ctx.fillStyle = "rgba(241, 78, 84, 0.15)"; ctx.fillRect(WORLD_SIZE-BASE_SIZE-camera.x, WORLD_SIZE-BASE_SIZE-camera.y, BASE_SIZE, BASE_SIZE);
}

ctx.fillStyle = "rgba(0,0,0,0.02)";
ctx.fillRect(1700 - camera.x, 1700 - camera.y, 600, 600);

const activeIds = gameState.entities.map(e => e.id);
for (let id of renderEntities.keys()) {
if (!activeIds.includes(id)) renderEntities.delete(id);
}

[...gameState.entities].sort((a,b) => (a.type.includes('tank')||a.type==='ai'?1:-1)).forEach(en => {
if (!renderEntities.has(en.id)) renderEntities.set(en.id, { x: en.x, y: en.y });
let rPos = renderEntities.get(en.id);
rPos.x += (en.x - rPos.x) * 0.35;
rPos.y += (en.y - rPos.y) * 0.35;

const sx = rPos.x - camera.x;
const sy = rPos.y - camera.y;

if(sx < -100 || sx > canvas.width+100 || sy < -100 || sy > canvas.height+100) return;

if(['tank','ai'].includes(en.type)) {
ctx.fillStyle = "white"; ctx.strokeStyle = "black"; ctx.lineWidth = 3;
ctx.font = "bold 14px Ubuntu"; ctx.textAlign = "center";
ctx.strokeText(en.name, sx, sy - en.radius - 25); ctx.fillText(en.name, sx, sy - en.radius - 25);
ctx.font = "11px Ubuntu";
ctx.strokeText(Math.floor(en.score), sx, sy - en.radius - 12); ctx.fillText(formatScore(Math.floor(en.score)), sx, sy - en.radius - 12);
}

if(en.hp < en.maxHp) {
ctx.fillStyle = '#555'; ctx.fillRect(sx-20, sy+en.radius+10, 40, 6);
ctx.fillStyle = '#85e37d'; ctx.fillRect(sx-20, sy+en.radius+10, 40*(en.hp/en.maxHp), 6);
}

ctx.save(); ctx.translate(sx, sy); ctx.lineWidth = 4;

if(['tank','ai'].includes(en.type)) {
let col = getTeamColor(en.team);

if (en.tankType === 'Smasher') {
ctx.save();
ctx.rotate(en.angle + Date.now() / 300);
ctx.fillStyle = "#555";
ctx.strokeStyle = darkenColor("#555", 30);
drawPoly(6, en.radius * 1.4);
ctx.restore();
}

ctx.fillStyle = col; ctx.strokeStyle = darkenColor(col, 30);

let specs = TANK_SPECS[en.tankType] || TANK_SPECS['Basic'];
specs.barrels.forEach(b => {
ctx.save(); ctx.rotate(en.angle + b.angle);
ctx.fillStyle = "#999"; ctx.strokeStyle = darkenColor("#999", 30);

if(b.w2) {
ctx.beginPath(); ctx.moveTo(0, -b.w/2); ctx.lineTo(en.radius * b.l, -b.w2/2);
ctx.lineTo(en.radius * b.l, b.w2/2); ctx.lineTo(0, b.w/2); ctx.closePath();
ctx.fill(); ctx.stroke();
} else {
ctx.fillRect(0, -b.w/2 + (b.y||0), en.radius * b.l, b.w);
ctx.strokeRect(0, -b.w/2 + (b.y||0), en.radius * b.l, b.w);
}
ctx.restore();
});

ctx.beginPath(); ctx.arc(0, 0, en.radius, 0, Math.PI*2); ctx.fill(); ctx.stroke();
} else {
if(en.type==='square'){ ctx.fillStyle=COLORS.square; ctx.strokeStyle=darkenColor(COLORS.square, 30); ctx.fillRect(-12,-12,24,24); ctx.strokeRect(-12,-12,24,24); }
if(en.type==='triangle'){ ctx.fillStyle=COLORS.triangle; ctx.strokeStyle=darkenColor(COLORS.triangle, 30); ctx.rotate(en.angle); drawPoly(3, en.radius); }
if(en.type==='pentagon'){ ctx.fillStyle=COLORS.pentagon; ctx.strokeStyle=darkenColor(COLORS.pentagon, 30); ctx.rotate(en.angle); drawPoly(5, en.radius); }
if(en.type==='hexagon'){ ctx.fillStyle=COLORS.hexagon; ctx.strokeStyle=darkenColor(COLORS.hexagon, 30); ctx.rotate(en.angle); drawPoly(6, en.radius); }
}
ctx.restore();
});

gameState.drones.forEach(d => {

if(sx < -50 || sx > canvas.width+50 || sy < -50 || sy > canvas.height+50) return;
ctx.save(); ctx.translate(sx, sy); ctx.rotate(d.angle);
ctx.lineWidth = 3; ctx.fillStyle = getTeamColor(d.team); ctx.strokeStyle = darkenColor(ctx.fillStyle, 30);
ctx.beginPath(); ctx.lineTo(d.radius, 0); ctx.lineTo(-d.radius*0.8, d.radius*0.8); ctx.lineTo(-d.radius*0.8, -d.radius*0.8); ctx.closePath();
ctx.fill(); ctx.stroke(); ctx.restore();
});

gameState.bullets.forEach(b => {
ctx.fillStyle = getTeamColor(b.team); ctx.strokeStyle = darkenColor(ctx.fillStyle, 30); ctx.lineWidth = 2;
ctx.beginPath(); ctx.arc(b.x-camera.x, b.y-camera.y, b.r, 0, Math.PI*2); ctx.fill(); ctx.stroke();
});

let scores = gameState.entities.filter(e => ['tank','ai'].includes(e.type)).sort((a,b) => b.score - a.score).slice(0, 8);
document.getElementById('score-list').innerHTML = scores.map(s => `<div style="color:${getTeamColor(s.team)}; margin: 2px 0;">${s.name}: ${Math.floor(s.score)}</div>`).join('');

mCtx.clearRect(0,0,150,150);
if(gameMode === "2TDM") {

requestAnimationFrame(draw);
}
