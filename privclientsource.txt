function switchTab(tab, btnElement = null) {
    document.querySelectorAll('.shop-tab').forEach(b => b.classList.remove('active'));
    
    // Safely apply active class whether called from code or a mouse click
    if (btnElement) {
        btnElement.classList.add('active');
    } else {
        const matchingBtn = Array.from(document.querySelectorAll('.shop-tab')).find(b => b.getAttribute('onclick').includes(tab));
        if (matchingBtn) matchingBtn.classList.add('active');
    }
    
    renderShopContent(tab);
}

function renderShopContent(category) {
    const container = document.getElementById('shop-content');
    container.innerHTML = '';
    
    if (shopItems[category].length === 0) {
        container.innerHTML = '<p style="text-align:center; color:#888; font-weight:bold;">Coming Soon!</p>';
        return;
    }

    let unlocked = JSON.parse(myProfile.unlocked_colors);
    
    shopItems[category].forEach(item => {
        let isUnlocked = unlocked.includes(item.id);
        let isEquipped = myProfile.selected_color === item.id;
        
        let btnHtml = '';
        if (isEquipped) {
            btnHtml = `<button class="shop-btn btn-equipped" disabled>Equipped</button>`;
        } else if (isUnlocked) {
            btnHtml = `<button class="shop-btn btn-equip" onclick="ws.send(JSON.stringify({type:'equip_item', category:'${category}', itemId:'${item.id}'}))">Equip</button>`;
        } else {
            let canAfford = myProfile.coins >= item.cost;
            btnHtml = `<button class="shop-btn btn-buy" ${!canAfford ? 'disabled' : ''} onclick="ws.send(JSON.stringify({type:'buy_item', category:'${category}', itemId:'${item.id}'}))">Buy (🪙 ${item.cost})</button>`;
        }

        container.innerHTML += `
            <div class="shop-item">
                <span style="color: ${item.id}; font-weight:bold; font-size: 16px;">${item.name}</span>
                ${btnHtml}
            </div>
        `;
    });
}

function formatScore(num) {
    if (num < 1000) return num.toString();
    if (num < 1000000) return (num / 1000).toFixed(num < 10000 ? 1 : 0) + "k";
    return (num / 1000000).toFixed(2).replace(/\.00$/, '') + "m";
}

function darkenColor(hex, percent) {
    hex = hex.replace('#', '');
    if (hex.length === 3) hex = hex[0]+hex[0]+hex[1]+hex[1]+hex[2]+hex[2];
    let r = parseInt(hex.substring(0, 2), 16);
    let g = parseInt(hex.substring(2, 4), 16);
    let b = parseInt(hex.substring(4, 6), 16);
    r = Math.max(0, Math.floor(r * (100 - percent) / 100));
    g = Math.max(0, Math.floor(g * (100 - percent) / 100));
    b = Math.max(0, Math.floor(b * (100 - percent) / 100));
    return "#" + (1 << 24 | r << 16 | g << 8 | b).toString(16).slice(1);
}

function connectWS(regionStr, modeStr) {
    const protocol = window.location.protocol === 'https:' ? 'wss://' : 'ws://';
    ws = new WebSocket(protocol + window.location.host);

    ws.onopen = () => {
        ws.send(JSON.stringify({ type: 'spectate', mode: modeStr, region: regionStr }));
        document.getElementById('playBtn').innerText = "Play";
        document.getElementById('playBtn').disabled = false;
        if (sessionToken) {
            ws.send(JSON.stringify({ type: 'auth_login', token: sessionToken }));
        }
        if (window.pingInterval) clearInterval(window.pingInterval);
        window.pingInterval = setInterval(() => { if(ws.readyState === 1) ws.send(JSON.stringify({ type: 'ping', time: Date.now() })); }, 1000);
        
        if (window.inputInterval) clearInterval(window.inputInterval);
        window.inputInterval = setInterval(() => {
            if(myId) {
                let finalAngle = Math.atan2(mouse.ry - (camera.y + canvas.height/2), mouse.rx - (camera.x + canvas.width/2));
                if (autoSpin) { spinAngle += 0.08; finalAngle = spinAngle; }

                ws.send(JSON.stringify({
                    type: 'input', up: keys.w || keys.arrowup, down: keys.s || keys.arrowdown,
                    left: keys.a || keys.arrowleft, right: keys.d || keys.arrowright,
                    shooting: mouse.pressed || autoFire, angle: finalAngle,
                    repel: mouse.repel, mouseX: mouse.rx, mouseY: mouse.ry
                }));
            }
        }, 1000 / TICK_RATE);
        const urlParams = new URLSearchParams(window.location.search);
        const token = urlParams.get('token');
        
        if (token) {
            ws.send(JSON.stringify({ type: 'auth_login', token: token }));
            window.history.replaceState({}, document.title, "/"); 
        }
    };

    ws.onmessage = (e) => {
        const data = JSON.parse(e.data);
        if(data.type === 'init') { myId = data.id; myTeam = data.team; spectateId = null; }
        else if(data.type === 'pong') { document.getElementById('ping-display').innerText = 'Ping: ' + (Date.now() - data.time) + ' ms'; }
        else if(data.type === 'state') { 
            gameState = data; 
            if (!window.drawing) { window.drawing = true; requestAnimationFrame(draw); }
        }
        else if(data.type === 'playerStats') { myStats = data; updateUI(); checkUpgrades(); }
        else if(data.type === 'spectate_update') { spectateId = data.id; }
        else if(data.type === 'death') {
            myId = null;
            spectateId = data.killerId; 
            document.getElementById('death-screen').style.display = 'flex';
            document.getElementById('ds-level').innerText = data.level + " " + data.tank;
            document.getElementById('ds-score').innerText = Math.floor(data.score);
            let m = Math.floor(data.timeAlive / 60); let s = data.timeAlive % 60;
            document.getElementById('ds-time').innerText = `${m}m ${s}s`;
            
            let dCanvas = document.getElementById('ds-tank-icon');
            let dctx = dCanvas.getContext('2d');
            dctx.clearRect(0,0,100,100);
            let col = getTeamColor(myTeam);
            
            dctx.fillStyle = col; dctx.strokeStyle = darkenColor(col, 30); dctx.lineWidth = 4;
            dctx.save(); dctx.translate(50, 50); dctx.rotate(-Math.PI/4); 
            
            let specs = TANK_SPECS[data.tank] || TANK_SPECS['Basic'];
            let r = 24; 

            specs.barrels.forEach(b => {
                dctx.save(); dctx.rotate(b.angle);
                dctx.fillStyle = "#999"; dctx.strokeStyle = darkenColor("#999", 30);
                if(b.w2) {
                    dctx.beginPath(); dctx.moveTo(0, -b.w/2); dctx.lineTo(r * b.l, -b.w2/2);
                    dctx.lineTo(r * b.l, b.w2/2); dctx.lineTo(0, b.w/2); dctx.closePath();
                    dctx.fill(); dctx.stroke();
                } else {
                    dctx.fillRect(0, -b.w/2 + (b.y||0), r * b.l, b.w);
                    dctx.strokeRect(0, -b.w/2 + (b.y||0), r * b.l, b.w);
                }
                dctx.restore();
            });

            if (data.tank === 'Smasher') {
                dctx.fillStyle = "#555"; dctx.strokeStyle = darkenColor("#555", 30);
                dctx.beginPath();
                for(let i=0; i<6; i++){
                    let a = (i * Math.PI * 2 / 6);
                    dctx.lineTo(Math.cos(a)*r*1.4, Math.sin(a)*r*1.4);
                }
                dctx.closePath(); dctx.fill(); dctx.stroke();
                dctx.fillStyle = col; dctx.strokeStyle = darkenColor(col, 30);
            }

            dctx.beginPath(); dctx.arc(0, 0, r, 0, Math.PI*2); dctx.fill(); dctx.stroke();
            dctx.restore();
        }
        if (data.type === 'profile_update') {
            myProfile = data.profile;
            updateAccountUI();
        }
    };

    ws.onclose = () => {
        if(myId) {
            document.getElementById('death-screen').style.display = 'flex';
            document.querySelector('.ds-title').innerText = "Disconnected";
            document.getElementById('ds-level').innerText = "Connection lost.";
            document.getElementById('ds-score').innerText = "--";
            document.getElementById('ds-time').innerText = "--";
        }
        document.getElementById('playBtn').innerText = "Connecting...";
        document.getElementById('playBtn').disabled = true;
    };
}

let currentUpgradesShown = "";
function checkUpgrades() {
    const upPanel = document.getElementById('upgrades-panel');
    let options = [];
    
    if (myStats.level >= 15 && myStats.tankType === 'Basic') {
        options = ['Twin', 'Sniper', 'Machine Gun', 'Flank Guard', 'Overseer', 'Smasher'];
    }

    let neededStr = options.join(",");
    if (currentUpgradesShown === neededStr) return;
    currentUpgradesShown = neededStr;
    
    upPanel.innerHTML = '';
    options.forEach(c => {
        let btn = document.createElement('div'); btn.className = 'upgrade-btn'; btn.innerText = c;
        btn.onclick = () => { 
            ws.send(JSON.stringify({ type: 'upgradeTank', tank: c })); 
            currentUpgradesShown = "";
            upPanel.innerHTML = '';
        };
        upPanel.appendChild(btn);
    });
}

function updateUI() {
    document.getElementById('score-text-overlay').innerText = "Score: " + Math.floor(myStats.score);
    document.getElementById('xp-bar').style.width = myStats.level >= 30 ? "100%" : ((myStats.xp / (myStats.level * 40)) * 100 + "%");
    document.getElementById('level-text').innerText = `Lvl ${myStats.level} ${myStats.tankType}`;
    
    let panel = document.getElementById('stats-panel');
    if (myStats.statPoints > 0) {
        panel.style.display = 'block';
        document.getElementById('stat-points').innerText = myStats.statPoints;
    } else { panel.style.display = 'none'; }

    myStats.stats.forEach((val, i) => {
        let slots = document.getElementById(`slots-${i}`).children;
        for(let j=0; j<7; j++) { slots[j].style.background = j < val ? STAT_INFO[i].color : '#555'; }
    });
}

function getTeamColor(team) {
    if(team === 0) return "#fff";
    if(gameMode === "FFA") return team === myTeam ? COLORS.team1 : COLORS.team2;
    return team === 1 ? COLORS.team1 : COLORS.team2;
}

function drawPoly(sides, r) {
    ctx.beginPath();
    for(let i=0; i<sides; i++){
        let a = (i * Math.PI * 2 / sides);
        ctx.lineTo(Math.cos(a)*r*1.2, Math.sin(a)*r*1.2);
    }
    ctx.closePath(); ctx.fill(); ctx.stroke();
}

function draw() {
    ctx.fillStyle = COLORS.bg; ctx.fillRect(0,0,canvas.width,canvas.height);

    let targetCamX = WORLD_SIZE/2 - canvas.width/2;
    let targetCamY = WORLD_SIZE/2 - canvas.height/2;

    if (myId) {
        let me = gameState.entities.find(e => e.id === myId);
        if(me) {
            if (!renderEntities.has(me.id)) renderEntities.set(me.id, { x: me.x, y: me.y });
            let rMe = renderEntities.get(me.id);
            targetCamX = rMe.x - canvas.width/2;
            targetCamY = rMe.y - canvas.height/2;

            if (myStats.tankType === 'Sniper' && (mouse.rightDown || keys.shift)) {
                let mouseAngle = Math.atan2(mouse.ry - rMe.y, mouse.rx - rMe.x);
                targetCamX += Math.cos(mouseAngle) * 450;
                targetCamY += Math.sin(mouseAngle) * 450;
            }
        }
    } else if (spectateId) {
        let spec = gameState.entities.find(e => e.id === spectateId);
        if(spec) {
            if (!renderEntities.has(spec.id)) renderEntities.set(spec.id, { x: spec.x, y: spec.y });
            let rSpec = renderEntities.get(spec.id);
            targetCamX = rSpec.x - canvas.width/2;
            targetCamY = rSpec.y - canvas.height/2;
        }
    }

    camera.x += (targetCamX - camera.x) * 0.1;
    camera.y += (targetCamY - camera.y) * 0.1;

    mouse.rx = mouse.x + camera.x; 
    mouse.ry = mouse.y + camera.y;

    ctx.strokeStyle = "rgba(0,0,0,0.06)"; 
    ctx.lineWidth = 1;
    ctx.beginPath();
    for(let x=-camera.x%50; x<canvas.width; x+=50) { ctx.moveTo(x,0); ctx.lineTo(x,canvas.height); }
    for(let y=-camera.y%50; y<canvas.height; y+=50) { ctx.moveTo(0,y); ctx.lineTo(canvas.width,y); }
    ctx.stroke();

    if(gameMode === "2TDM") {
        ctx.fillStyle = "rgba(0, 178, 225, 0.15)"; ctx.fillRect(-camera.x, -camera.y, BASE_SIZE, BASE_SIZE);
        ctx.fillStyle = "rgba(241, 78, 84, 0.15)"; ctx.fillRect(WORLD_SIZE-BASE_SIZE-camera.x, WORLD_SIZE-BASE_SIZE-camera.y, BASE_SIZE, BASE_SIZE);
    }

    ctx.fillStyle = "rgba(0,0,0,0.02)"; 
    ctx.fillRect(1700 - camera.x, 1700 - camera.y, 600, 600);

    const activeIds = gameState.entities.map(e => e.id);
    for (let id of renderEntities.keys()) {
        if (!activeIds.includes(id)) renderEntities.delete(id);
    }

    [...gameState.entities].sort((a,b) => (a.type.includes('tank')||a.type==='ai'?1:-1)).forEach(en => {
        if (!renderEntities.has(en.id)) renderEntities.set(en.id, { x: en.x, y: en.y });
        let rPos = renderEntities.get(en.id);
        rPos.x += (en.x - rPos.x) * 0.35;
        rPos.y += (en.y - rPos.y) * 0.35;

        const sx = rPos.x - camera.x; 
        const sy = rPos.y - camera.y;

        if(sx < -100 || sx > canvas.width+100 || sy < -100 || sy > canvas.height+100) return;

        if(['tank','ai'].includes(en.type)) {
            ctx.fillStyle = en.nameColor || "white"; ctx.strokeStyle = "black"; ctx.lineWidth = 3;
            ctx.font = "bold 14px Ubuntu"; ctx.textAlign = "center";
            ctx.strokeText(en.name, sx, sy - en.radius - 25); ctx.fillText(en.name, sx, sy - en.radius - 25);
            ctx.font = "11px Ubuntu";
            ctx.strokeText(Math.floor(en.score), sx, sy - en.radius - 12); ctx.fillText(formatScore(Math.floor(en.score)), sx, sy - en.radius - 12);
        }

        if(en.hp < en.maxHp) {
            ctx.fillStyle = '#555'; ctx.fillRect(sx-20, sy+en.radius+10, 40, 6);
            ctx.fillStyle = '#85e37d'; ctx.fillRect(sx-20, sy+en.radius+10, 40*(en.hp/en.maxHp), 6);
        }

        ctx.save(); ctx.translate(sx, sy); ctx.lineWidth = 4;

        if(['tank','ai'].includes(en.type)) {
            let col = getTeamColor(en.team);
            
            if (en.tankType === 'Smasher') {
                ctx.save();
                ctx.rotate(en.angle + Date.now() / 300); 
                ctx.fillStyle = "#555";
                ctx.strokeStyle = darkenColor("#555", 30);
                drawPoly(6, en.radius * 1.4);
                ctx.restore();
            }

            ctx.fillStyle = col; ctx.strokeStyle = darkenColor(col, 30); 
            
            let specs = TANK_SPECS[en.tankType] || TANK_SPECS['Basic'];
            specs.barrels.forEach(b => {
                ctx.save(); ctx.rotate(en.angle + b.angle);
                ctx.fillStyle = "#999"; ctx.strokeStyle = darkenColor("#999", 30); 

                if(b.w2) {
                    ctx.beginPath(); ctx.moveTo(0, -b.w/2); ctx.lineTo(en.radius * b.l, -b.w2/2);
                    ctx.lineTo(en.radius * b.l, b.w2/2); ctx.lineTo(0, b.w/2); ctx.closePath();
                    ctx.fill(); ctx.stroke();
                } else {
                    ctx.fillRect(0, -b.w/2 + (b.y||0), en.radius * b.l, b.w);
                    ctx.strokeRect(0, -b.w/2 + (b.y||0), en.radius * b.l, b.w);
                }
                ctx.restore();
            });

            ctx.beginPath(); ctx.arc(0, 0, en.radius, 0, Math.PI*2); ctx.fill(); ctx.stroke();
        } else {
            if(en.type==='square'){ ctx.fillStyle=COLORS.square; ctx.strokeStyle=darkenColor(COLORS.square, 30); ctx.fillRect(-12,-12,24,24); ctx.strokeRect(-12,-12,24,24); }
            if(en.type==='triangle'){ ctx.fillStyle=COLORS.triangle; ctx.strokeStyle=darkenColor(COLORS.triangle, 30); ctx.rotate(en.angle); drawPoly(3, en.radius); }
            if(en.type==='pentagon'){ ctx.fillStyle=COLORS.pentagon; ctx.strokeStyle=darkenColor(COLORS.pentagon, 30); ctx.rotate(en.angle); drawPoly(5, en.radius); }
            if(en.type==='hexagon'){ ctx.fillStyle=COLORS.hexagon; ctx.strokeStyle=darkenColor(COLORS.hexagon, 30); ctx.rotate(en.angle); drawPoly(6, en.radius); }
        }
        ctx.restore();
    });

    gameState.drones.forEach(d => {
        const sx = d.x - camera.x; const sy = d.y - camera.y;
        if(sx < -50 || sx > canvas.width+50 || sy < -50 || sy > canvas.height+50) return;
        ctx.save(); ctx.translate(sx, sy); ctx.rotate(d.angle);
        ctx.lineWidth = 3; ctx.fillStyle = getTeamColor(d.team); ctx.strokeStyle = darkenColor(ctx.fillStyle, 30);
        ctx.beginPath(); ctx.lineTo(d.radius, 0); ctx.lineTo(-d.radius*0.8, d.radius*0.8); ctx.lineTo(-d.radius*0.8, -d.radius*0.8); ctx.closePath();
        ctx.fill(); ctx.stroke(); ctx.restore();
    });

    gameState.bullets.forEach(b => {
        ctx.fillStyle = getTeamColor(b.team); ctx.strokeStyle = darkenColor(ctx.fillStyle, 30); ctx.lineWidth = 2;
        ctx.beginPath(); ctx.arc(b.x-camera.x, b.y-camera.y, b.r, 0, Math.PI*2); ctx.fill(); ctx.stroke();
    });

    let scores = gameState.entities.filter(e => ['tank','ai'].includes(e.type)).sort((a,b) => b.score - a.score).slice(0, 8);
    document.getElementById('score-list').innerHTML = scores.map(s => `<div style="color:${getTeamColor(s.team)}; margin: 2px 0;">${s.name}: ${Math.floor(s.score)}</div>`).join('');

    mCtx.clearRect(0,0,150,150);
    if(gameMode === "2TDM") {
        mCtx.fillStyle = "rgba(0, 178, 225, 0.3)"; mCtx.fillRect(0, 0, (BASE_SIZE/WORLD_SIZE)*150, (BASE_SIZE/WORLD_SIZE)*150);
        mCtx.fillStyle = "rgba(241, 78, 84, 0.3)"; mCtx.fillRect(150-(BASE_SIZE/WORLD_SIZE)*150, 150-(BASE_SIZE/WORLD_SIZE)*150, (BASE_SIZE/WORLD_SIZE)*150, (BASE_SIZE/WORLD_SIZE)*150);
    }
    gameState.entities.forEach(en => {
        if(!['tank','ai'].includes(en.type)) return;
        mCtx.fillStyle = en.id === myId ? '#fff' : getTeamColor(en.team);
        let s = en.id === myId ? 4 : 2;
        mCtx.fillRect((en.x/WORLD_SIZE)*150 - s/2, (en.y/WORLD_SIZE)*150 - s/2, s, s);
    });

    requestAnimationFrame(draw);
}

window.onkeydown = e => { 
    let key = e.key.toLowerCase();
    if(key === 'shift') { mouse.repel = true; keys.shift = true; } 
    keys[key] = true; 

    if(key === 'e') autoFire = !autoFire;
    if(key === 'c') autoSpin = !autoSpin;

    if(key >= '1' && key <= '8') {
        if(ws && ws.readyState === 1) {
            ws.send(JSON.stringify({ type: 'upgradeStat', statIndex: parseInt(key) - 1 }));
        }
    }
};

window.onkeyup = e => { 
    let key = e.key.toLowerCase();
    if(key === 'shift') { mouse.repel = false; keys.shift = false; } 
    keys[key] = false; 
};

window.onmousemove = e => { 
    mouse.x = e.clientX; mouse.y = e.clientY; 
    mouse.rx = mouse.x + camera.x; mouse.ry = mouse.y + camera.y; 
};

window.onmousedown = (e) => {
    if(e.button === 0) mouse.pressed = true;
    if(e.button === 2) mouse.rightDown = true;
};
window.onmouseup = (e) => {
    if(e.button === 0) mouse.pressed = false;
    if(e.button === 2) mouse.rightDown = false;
};
window.oncontextmenu = e => e.preventDefault(); 
window.onresize = () => { canvas.width = window.innerWidth; canvas.height = window.innerHeight; };
canvas.width = window.innerWidth; canvas.height = window.innerHeight;
